    <script>
        // Matrix Digital Rain
        const canvas = document.getElementById('matrix-canvas');
        const ctx = canvas.getContext('2d');

        function resizeCanvas() {
            canvas.width = window.innerWidth;
            canvas.height = window.innerHeight;
        }
        window.addEventListener('resize', resizeCanvas);
        resizeCanvas();

        const chars = 'ABCDEFGHIJKLMNOPQRSTUVWXYZ0123456789@#$%^&*]*ｦｧｨｩｪｫｬｭｮｯｰｱｲｳｴｵｶｷｸｹｺｻｼｽｾｿﾀﾁﾂﾃﾄﾅﾆﾇﾈﾉﾊﾋﾌﾍﾎﾏﾐﾑﾒﾓﾔﾕﾖﾗﾘﾙﾚﾛﾜﾝ'.split('');
        const fontSize = 16;
        let columns = Math.floor(canvas.width / fontSize);
        let drops = [];

        // Initialisation des gouttes
        for (let x = 0; x < columns; x++) {
            drops[x] = Math.random() * (canvas.height / fontSize);
        }

        window.addEventListener('resize', () => {
            const newColumns = Math.floor(canvas.width / fontSize);
            if (newColumns > columns) {
                for (let x = columns; x < newColumns; x++) {
                    drops[x] = Math.random() * (canvas.height / fontSize);
                }
            }
            columns = newColumns;
        });

        function draw() {
            // Fond semi-transparent pour l'effet de traînée
            ctx.fillStyle = 'rgba(0, 0, 0, 0.05)';
            ctx.fillRect(0, 0, canvas.width, canvas.height);

            // Texte Vert Matrix
            ctx.fillStyle = '#00ff41';
            ctx.font = fontSize + 'px monospace';

            for (let i = 0; i < drops.length; i++) {
                const text = chars[Math.floor(Math.random() * chars.length)];

                // Effet de surbrillance aléatoire sur certains caractères
                if (Math.random() > 0.95) {
                    ctx.fillStyle = '#fff';
                } else {
                    ctx.fillStyle = '#00ff41';
                }

                ctx.fillText(text, i * fontSize, drops[i] * fontSize);

                // Réinitialisation de la goutte avec un peu d'aléatoire
                if (drops[i] * fontSize > canvas.height && Math.random() > 0.975) {
                    drops[i] = 0;
                }
                drops[i]++;
            }
        }

        // Vitesse d'animation (33ms = ~30fps)
        setInterval(draw, 33);

        // Scroll Reveal Animation (Intersection Observer ou calcul simple)
        function reveal() {
            var reveals = document.querySelectorAll(".reveal");
            for (var i = 0; i < reveals.length; i++) {
                var windowHeight = window.innerHeight;
                var elementTop = reveals[i].getBoundingClientRect().top;
                var elementVisible = 120; // Déclenchement avant l'apparition

                if (elementTop < windowHeight - elementVisible) {
                    reveals[i].classList.add("active");
                }
            }
        }
        window.addEventListener("scroll", reveal);
        // Déclencher une fois au chargement pour les éléments déjà visibles
        setTimeout(reveal, 300); 

        // Terminal Loader Sequence (Integrated in Header)
        document.addEventListener("DOMContentLoaded", async () => {
            document.body.style.overflow = "hidden"; // Prevent scrolling during loading
            window.scrollTo(0, 0); // Force scroll to top
            
            const lines = [
                "> v 26.27",
                "> ACCÈS AUTORISÉ : DÉBUT DE TRANSMISSION",
                "> nano PROGRAMME_MATRIX_COMPLET.md"
            ];
            
            const loaderContent = document.getElementById("loader-content");
            const headerTerminal = document.getElementById("header-terminal");
            const restOfHeader = document.getElementById("rest-of-header");
            
            const typeLetter = (text, element) => {
                return new Promise(resolve => {
                    let i = 0;
                    // Typing speed per letter
                    const speed = Math.random() * 15 + 10; 
                    const interval = setInterval(() => {
                        element.textContent += text.charAt(i);
                        i++;
                        if (i >= text.length) {
                            clearInterval(interval);
                            resolve();
                        }
                    }, speed);
                });
            };

            for (let i = 0; i < lines.length; i++) {
                const p = document.createElement("p");
                loaderContent.appendChild(p);
                await typeLetter(lines[i], p);
                // Pause slightly between lines
                await new Promise(r => setTimeout(r, Math.random() * 100 + 50));
            }

            // End of sequence: fade in the rest of the header and dim the terminal text to match design
            setTimeout(() => {
                headerTerminal.classList.remove("text-neon-green", "opacity-100");
                headerTerminal.classList.add("text-neon-dark", "opacity-70");
                
                restOfHeader.classList.remove("opacity-0", "pointer-events-none");
                restOfHeader.classList.add("opacity-100", "pointer-events-auto");
                
                document.body.style.overflow = ""; // Restore scrolling
                
                // Wait for fade in, then trigger reveal for any other elements
                setTimeout(reveal, 800); 
            }, 300);
        });
    </script>
